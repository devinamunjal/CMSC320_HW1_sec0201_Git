
import requests
import pandas as pd
api_url = "https://www.ngdc.noaa.gov/hazel/hazard-service/api/v1/volcanoes"
response = requests.get(api_url)
data = response.json()
df = pd.DataFrame(data["items"])
df.to_csv("volcanoes.csv", index=False)
